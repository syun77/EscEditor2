def knob
{
	$50 = 0
	ITEM_HAS(ITEM_YELLOW_KEY)
	if($1) {
		$50 += 1
	}
	ITEM_HAS(ITEM_BLUE_KEY)
	if($1) {
		$50 += 1
	}
	ITEM_HAS(ITEM_GREEN_KEY)
	if($1) {
		$50 += 1
	}
	
	if($50 == 3) {
		%50 = true
		":ドアが開いた"
	}
	else if($50 > 0) {
		$51 = 3 - $50
		":カギがあと$V51本必要だ"
	}
	else {
		":カギがかかっていて扉は開かない"
	}
}
